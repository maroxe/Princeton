  import os

  import glob
  def createlinkstoall(ext):
      def linkto(s):
          parts = s.split('/')
          return (s, parts[-3], parts[-1].split('.')[0])
      return map(lambda s: ['[[./%s][%s - %s]]' % (linkto(s))], glob.glob('*/*/*.%s' % ext))
